Массивы в C++

Массивы позволяют хранить набор значений одного типа в непрерывной области памяти. В C++ существуют обычные массивы (C-style arrays) и более современные контейнеры из STL (например, vector).

Одномерные массивы

Объявление и инициализация одномерного массива:

// Объявление массива из 5 целых чисел
int numbers[5];

// Инициализация при объявлении
int primes[] = {2, 3, 5, 7, 11};

// Доступ к элементам массива
primes[0] = 1;  // Изменяем первый элемент
int x = primes[2];  // x = 5

Пример работы с массивом

#include <iostream>

int main() {
    const int SIZE = 5;
    double temperatures[SIZE] = {20.5, 22.0, 18.5, 21.3, 19.8};
    
    // Вывод всех элементов массива
    for (int i = 0; i < SIZE; i++) {
        std::cout << "Температура в день " << i+1 << ": " << temperatures[i] << std::endl;
    }
    
    // Вычисление средней температуры
    double sum = 0;
    for (int i = 0; i < SIZE; i++) {
        sum += temperatures[i];
    }
    double average = sum / SIZE;
    std::cout << "Средняя температура: " << average << std::endl;
    
    return 0;
}

Многомерные массивы

C++ поддерживает массивы с несколькими измерениями:

// Двумерный массив (матрица)
int matrix[3][3] = {
    {1, 2, 3},
    {4, 5, 6},
    {7, 8, 9}
};

// Трехмерный массив
int cube[2][2][2] = {
    {
        {1, 2},
        {3, 4}
    },
    {
        {5, 6},
        {7, 8}
    }
};

Пример работы с матрицей

#include <iostream>

int main() {
    const int ROWS = 3, COLS = 3;
    int matrix[ROWS][COLS] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };
    
    // Вывод матрицы
    for (int i = 0; i < ROWS; i++) {
        for (int j = 0; j < COLS; j++) {
            std::cout << matrix[i][j] << " ";
        }
        std::cout << std::endl;
    }
    
    // Сумма элементов главной диагонали
    int diagonalSum = 0;
    for (int i = 0; i < ROWS; i++) {
        diagonalSum += matrix[i][i];
    }
    std::cout << "Сумма диагонали: " << diagonalSum << std::endl;
    
    return 0;
}

Динамические массивы

Для создания массивов переменного размера используются указатели и операторы new/delete:

#include <iostream>

int main() {
    int size;
    std::cout << "Введите размер массива: ";
    std::cin >> size;
    
    // Создание динамического массива
    int* dynamicArray = new int[size];
    
    // Заполнение массива
    for (int i = 0; i < size; i++) {
        dynamicArray[i] = i * 10;
    }
    
    // Вывод массива
    for (int i = 0; i < size; i++) {
        std::cout << dynamicArray[i] << " ";
    }
    std::cout << std::endl;
    
    // Освобождение памяти
    delete[] dynamicArray;
    
    return 0;
}

Совет по использованию массивов

В современном C++ предпочтительнее использовать контейнеры STL (например, std::vector или std::array), так как они безопаснее и предоставляют больше возможностей. C-style массивы стоит использовать только в особых случаях.

Массивы и указатели

Имя массива является указателем на его первый элемент:

int arr[5] = {10, 20, 30, 40, 50};
int* ptr = arr;  // ptr указывает на первый элемент массива

// Доступ через указатель
std::cout << *ptr << std::endl;  // 10
std::cout << *(ptr + 2) << std::endl;  // 30

// Арифметика указателей
ptr++;  // теперь указывает на второй элемент
std::cout << *ptr << std::endl;  // 20

Передача массива в функцию

#include <iostream>

// Функция принимает массив и его размер
void printArray(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;
}

int main() {
    int numbers[] = {1, 2, 3, 4, 5};
    printArray(numbers, 5);
    return 0;
}
